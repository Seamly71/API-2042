FROM python:3.13.7

RUN apt-get update -y
RUN apt-get install -y gdal-bin libgdal-dev

RUN apt-get install -y binutils libproj-dev

COPY requirements.txt .

COPY . .

CMD ["gunicorn", "--bind", "0.0.0.0:80", "API_2042.wsgi"]
